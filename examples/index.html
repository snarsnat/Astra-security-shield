ASTRA Shield - Usage Examples

1. Basic Usage

import { ASTRAShield } from 'astra-shield';

const shield = new ASTRAShield({ debug: true });

// Protect an action
async function login() {
  const result = await shield.protect('login', { userId: '123' });
  console.log(result);
}

2. Event Listeners

shield.on('success', (data) => {
  console.log('Verified!', data);
});

shield.on('challenge', (data) => {
  console.log('Challenge started:', data.tier);
});

shield.on('blocked', (data) => {
  console.log('Blocked:', data.reason);
});

3. Manual Verification

// Force verification at any time
const result = await shield.verify();
if (result.success) {
  // User is human
}

4. Accessibility Options

shield.accessibility.setPreference('reduceMotion', true);
shield.accessibility.setPreference('highContrast', true);
shield.accessibility.setPreference('extendedTime', true);

5. Custom Configuration

const shield = new ASTRAShield({
  apiKey: 'your-api-key',
  endpoint: '/api/verify',
  sessionDuration: 60 * 60 * 1000, // 1 hour
  mutationInterval: 60 * 60 * 1000, // 1 hour
  theme: 'auto'
});
